```sql
-- 011_press_releases_phase7_columns.sql
-- Adds Phase 7 columns to press_releases table created in migration 010.
-- Idempotent: uses ADD COLUMN IF NOT EXISTS (PostgreSQL 9.6+).

ALTER TABLE press_releases
  ADD COLUMN IF NOT EXISTS compliance_label TEXT,
  ADD COLUMN IF NOT EXISTS media_files JSONB,
  ADD COLUMN IF NOT EXISTS wire_results JSONB,
  ADD COLUMN IF NOT EXISTS research_context JSONB;

-- Also drop the wire_service single-value column if it exists (Phase 7 uses wire_results JSONB array instead)
-- Only drop if it was created as a constraint-limited single value in 010
-- Use a DO block to check existence first
DO $$
BEGIN
  IF EXISTS (
    SELECT 1 FROM information_schema.columns
    WHERE table_name = 'press_releases' AND column_name = 'wire_service'
  ) THEN
    ALTER TABLE press_releases DROP COLUMN wire_service;
  END IF;
  IF EXISTS (
    SELECT 1 FROM information_schema.columns
    WHERE table_name = 'press_releases' AND column_name = 'status'
    AND table_schema = 'public'
  ) THEN
    -- status column stays — it's still valid for overall PR status
    NULL;
  END IF;
END $$;
```
