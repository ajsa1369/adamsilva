```typescript
import { buildImageObjectSchema } from '@/lib/schemas/image'
import type { ImagePipelineInput, ImagePipelineResult } from '@/lib/blog/types'

// Slug-safe string helper: lowercase, spaces → hyphens, strip special chars
function toSlug(s: string): string {
  return s.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '')
}

// Generate N scene descriptions from topic data
// Uses topic.targetQueries and topic.faqClusters to produce varied scene labels
function generateSceneDescriptions(topic: ImagePipelineInput['topic'], count: number): string[] {
  const scenes: string[] = ['overview', 'diagram', 'infographic', 'comparison', 'workflow', 'architecture', 'results', 'summary']
  // Incorporate topic data into first few scenes
  const topicSlug = toSlug(topic.title)
  const baseScenes = [
    topicSlug + '-overview',
    topicSlug + '-diagram',
    ...scenes.slice(2),
  ]
  return baseScenes.slice(0, count)
}

export function generateImages(input: ImagePipelineInput): ImagePipelineResult[] {
  const topicSlug = toSlug(input.postTitle)
  const results: ImagePipelineResult[] = []

  const scenes = generateSceneDescriptions(input.topic, input.imageCount)

  for (let i = 0; i < input.imageCount; i++) {
    const scene = scenes[i] ?? `scene-${i + 1}`
    const filename = `${topicSlug}-${scene}-${i + 1}.png`
    const contentUrl = `${input.imagesBaseUrl}/${filename}`

    // TODO(v2): Replace stub with actual AI image generation API call
    // e.g. await openai.images.generate({ prompt: `${input.topic.title} - ${scene}`, ... })

    const jsonLd = buildImageObjectSchema({
      filename: contentUrl,  // pass full URL — buildImageObjectSchema detects http prefix
      name: `${input.postTitle} — ${scene.split('-').map(w => w.charAt(0).toUpperCase() + w.slice(1)).join(' ')}`,
      description: `${input.topic.title}: ${input.topic.targetQueries[0] ?? scene}. Authority gap score: ${input.topic.authorityGapScore}/100. Generated for ${input.clientName}.`,
      width: 1200,
      height: 630,
      keywords: input.topic.targetQueries.slice(0, 3).join(', '),
      about: { '@type': 'Article', name: input.postTitle },
      encodingFormat: 'image/png',
      representativeOfPage: i === 0,  // first image is representative
    })

    results.push({ filename, contentUrl, jsonLd })
  }

  return results
}
```

```typescript
import { spawn } from 'child_process'
import path from 'path'
import { SITE_URL, ORG_ID } from '@/lib/schemas/organization'
import type { VideoPipelineInput, VideoPipelineResult } from '@/lib/blog/types'

// Build voiceover transcript from topic data (Answer-First format + FAQ answers stub)
function buildTranscript(input: VideoPipelineInput): string {
  const intro = `${input.postTitle}. ${input.excerpt}`
  const faqLines = input.topic.faqClusters.slice(0, 3).map(q => `${q}`)
  return [intro, ...faqLines].join(' ')
}

// Parse ISO 8601 duration from frame count (30fps, 840 frames = PT28S)
function framesToDuration(frames: number, fps = 30): string {
  const seconds = Math.round(frames / fps)
  const minutes = Math.floor(seconds / 60)
  const remainingSeconds = seconds % 60
  if (minutes > 0) {
    return `PT${minutes}M${remainingSeconds}S`
  }
  return `PT${remainingSeconds}S`
}

// Spawn npx remotion render as child process
// Returns promise resolving to output file path
async function spawnRemotionRender(
  slug: string,
  props: Record<string, unknown>,
  outputPath: string
): Promise<void> {
  return new Promise((resolve, reject) => {
    const propsJson = JSON.stringify(props)
    const args = [
      'remotion',
      'render',
      'remotion/index.ts',
      'BlogSummary',
      `--props=${propsJson}`,
      `--output=${outputPath}`,
      '--log=verbose',
    ]

    // TODO(v2): For production, use @remotion/renderer programmatic API or Remotion Lambda
    // This CLI approach works for local/CI but blocks the Node.js thread during render
    const proc = spawn('npx', args, {
      stdio: 'pipe',
      shell: process.platform === 'win32',
    })

    const stderr: string[] = []
    proc.stderr?.on('data', (chunk: Buffer) => stderr.push(chunk.toString()))

    proc.on('close', (code) => {
      if (code === 0) {
        resolve()
      } else {
        reject(new Error(`Remotion render exited with code ${code}. stderr: ${stderr.join('')}`))
      }
    })

    proc.on('error', (err) => {
      reject(new Error(`Failed to spawn remotion render: ${err.message}`))
    })
  })
}

export async function generateVideo(input: VideoPipelineInput): Promise<VideoPipelineResult> {
  const outputDir = input.outputDir ?? 'public/videos/blog'
  const videoFilename = `${input.slug}.mp4`
  const thumbFilename = `${input.slug}-thumb.jpg`
  const videoPath = path.join(outputDir, videoFilename)
  const thumbnailPath = path.join(outputDir, thumbFilename)

  // BlogSummaryVideo props — maps topic faqClusters to keyPoints
  const remotionProps = {
    title: input.postTitle,
    excerpt: input.excerpt,
    keyPoints: input.topic.faqClusters.slice(0, 4),
    author: input.authorName,
    category: input.category ?? 'Agentic Commerce',
    slug: input.slug,
    protocols: [] as string[],
  }

  const transcript = buildTranscript(input)

  // Remotion BlogSummary: 30fps, 840 frames = PT28S
  const FRAMES = 840
  const FPS = 30
  const duration = framesToDuration(FRAMES, FPS)

  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL ?? SITE_URL
  const contentUrl = `${siteUrl}/videos/blog/${videoFilename}`
  const thumbnailUrl = `${siteUrl}/videos/blog/${thumbFilename}`
  const postUrl = `${siteUrl}/insights/${input.slug}`

  // Attempt render — gracefully stub on failure (CI / no display environment)
  try {
    await spawnRemotionRender(input.slug, remotionProps, videoPath)
  } catch (err) {
    // TODO(v2): surface this to cron error log; stub result in v1 allows pipeline to continue
    console.warn(`[video-pipeline] Remotion render failed (stub mode): ${err instanceof Error ? err.message : String(err)}`)
  }

  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'VideoObject',
    '@id': `${postUrl}#video`,
    name: `Video Summary: ${input.postTitle}`,
    description: input.excerpt,
    thumbnailUrl,
    uploadDate: new Date().toISOString(),
    duration,
    transcript,
    contentUrl,
    embedUrl: contentUrl,
    author: {
      '@type': 'Person',
      name: input.authorName,
      '@id': `${siteUrl}/#person`,
    },
    publisher: { '@id': ORG_ID },
    about: {
      '@type': 'Article',
      name: input.postTitle,
      url: postUrl,
    },
    inLanguage: 'en-US',
  }

  return {
    videoPath,
    thumbnailPath,
    transcript,
    duration,
    jsonLd,
  }
}
```

Key constraints:
- `child_process` spawn with `shell: process.platform === 'win32'` (Windows npx compatibility)
- Remotion render failure is caught and logged — pipeline continues with stub data so cron does not halt
- No `any` types — use `Record<string, unknown>` for remotionProps passed to spawn
- `NEXT_PUBLIC_SITE_URL` env fallback to SITE_URL constant
