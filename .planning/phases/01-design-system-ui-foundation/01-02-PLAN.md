```typescript
  // string: displayed as-is
  // null: displayed as dash
  highlight?: boolean       // bold row label
}

  columns: string[]         // column headers (e.g. ["Bronze", "Silver", "Gold", "Core"])
  rows: ComparisonRow[]
  caption?: string
  className?: string
}
```

Server component.

Layout:
- Outer: `overflow-x-auto w-full`
- `<table>` with `w-full text-sm border-collapse`
- `<thead>`: `bg-[var(--color-surface-2)]` with `text-[var(--color-text)] font-semibold text-center py-3 px-4`
- `<tbody>`: alternating row backgrounds: odd `bg-[var(--color-surface)]`, even `bg-transparent`
- Feature cell: left-aligned, `font-medium text-[var(--color-text)]` (bold if `highlight`)
- Value cell: centered
  - `true` → green check SVG or `✓` in `text-emerald-400 font-bold`
  - `false` → red cross or `✗` in `text-red-400`
  - `null` → `—` in `text-[var(--color-muted-2)]`
